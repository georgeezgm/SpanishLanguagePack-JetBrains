리시버에서 아무것도 액세스하지 않는 불필요한 with 함수 호출을 보고합니다.

예:


  class MyClass {
      fun f(): String = ""
  }

  fun testRedundant() {
      with(c) { // <== 'c'가 사용되지 않으므로 'with'는 불필요
          println("1")
      }
  }

  fun testOk() {
      val c = MyClass()
      with(c) { // <== 'f()'가 실질적으로 'c.f()'이므로 문제 없음
          println(f())
      }
  }

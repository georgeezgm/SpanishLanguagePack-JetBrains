Spring XML 애플리케이션 컨텍스트의 비활성 프로파일을 보고합니다.

예:


  <!-- 활성화된 프로필 'DEV'는 이 애플리케이션 컨텍스트용으로 선택되었습니다 -->
  <beans profile="DEV">
    <bean class="MyFactory" factory-method="createBean"/>
  </beans>
  <!--  이 태그의 내용은 'unused'로 강조 표시됩니다 -->
  <beans profile="TESTS">
    <bean class="MyTestFactory" factory-method="createBean"/>
  </beans>
  <!--  'unused' 섹션의 끝 -->
  </beans>

불필요한 import 문을 보고합니다.

기본 및 사용하지 않은 가져오기는 안전하게 제거할 수 있습니다.

예:


  import kotlin.*
  import kotlin.collections.*
  import kotlin.comparisons.*
  import kotlin.io.*
  import kotlin.ranges.*
  import kotlin.sequences.*
  import kotlin.text.*

  // jvm 전용
  import java.lang.*
  import kotlin.jvm.*

  // js 전용
  import kotlin.js.*

  import java.io.* // 이 가져오기는 사용하지 않으므로 제거해도 됨
  import java.util.*

  fun foo(list: ArrayList<String>) {
      list.add("")
  }

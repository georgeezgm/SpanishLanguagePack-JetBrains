관련 매핑이 없는 서블릿을 보고합니다.

예:


  <web-app>
    <servlet>
      <servlet-name>foo</servlet-name> // 오류: 서블릿에 매핑이 있어야 함
      <servlet-class>javax.servlet.Servlet</servlet-class>
    </servlet>
  
    <filter>
      <filter-name>bar</filter-name> // 오류: 필터에 매핑이 있어야 함
      <filter-class>java.lang.String</filter-class>
    </filter>
  </web-app>

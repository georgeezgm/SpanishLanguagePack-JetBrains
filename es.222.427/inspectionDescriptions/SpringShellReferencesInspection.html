Spring Shell org.springframework.shell.standard.ShellMethodAvailability 어노테이션 내부에 있는 잘못된 참조를 보고합니다.

예:


@ShellComponent
public class MyCommands {

    @ShellMethod("sum")
    @ShellMethodAvailability("sumAvailability") // 어노테이션 값이 미해결 심볼로 강조 표시됨
    public int sum(int a, int b) {
        return a + b;
    }

    @ShellMethodAvailability("sumMethod") // 어노테이션 값이 미해결 심볼로 강조 표시됨
    public Availability isSumAvailable() {
        return Availability.available();
    }
}

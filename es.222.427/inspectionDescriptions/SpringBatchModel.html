Spring Batch XML 기반 애플리케이션 컨텍스트 문제를 보고합니다.
  • 해결되지 않은 bean 참조
  • 태그 및 속성 누락
  • 잘못된 프로퍼티 타입
  • 일관성 없는 enum 프로퍼티
  • 참조된 bean의 잘못된 타입

예:


  <beans ... >
  <batch:job id="" <!-- 값이 비어 있으면 안 됩니다 -->
    incrementer="dummyBean"  <!-- Bean은 'org.springframework.batch.core.JobParametersIncrementer' 타입이어야 합니다 -->
    job-repository="transactionManager"  <!-- Bean은 'org.springframework.batch.core.repository.JobRepository' 타입이어야 합니다 -->
    parent="INVALID_VALUE" <!-- 잡 INVALID_VALUE를 해결할 수 없습니다 -->
    restartable="INVALID_VALUE"> <!-- 심볼 INVALID_VALUE를 해결할 수 없습니다 -->
    <batch:validator
      ref="dataSource">  <!-- Bean은 'org.springframework.batch.core.JobParametersValidator' 타입이어야 합니다 -->
    </batch:validator>
  </batch:job>
  <batch:job-repository/>

  <bean id="dataSource" class="org.springframework.jdbc.datasource.DelegatingDataSource"/>
  <bean id="transactionManager" class="org.springframework.jdbc.datasource.DataSourceTransactionManager"/>

  <bean id="dummyBean" class="java.lang.String"/>
</beans>

XML 애플리케이션 컨텍스트에 정의된 잘못된 schema bean을 보고합니다.


  <beans ... xmlns:util="https://www.springframework.org/schema/util">
    <bean class="MyBean" id="myBean"/>
    <bean class="ErrTypeBean" id="errTypeBean"/>
    <util:list list-class="java.lang.String"/> // "클래스가 'java.util.List'의 상속자여야 합니다"라고 보고됩니다
    <util:set value-type="MyBean">
      <ref bean="fooBean"/>
      <ref bean="errTypeBean"/> // " 'MyBean' 타입의 컬렉션에 추가할 수 없습니다"라고 보고됩니다
    </util:set>
  </beans>

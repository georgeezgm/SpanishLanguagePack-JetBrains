ClassVar 어노테이션의 잘못된 사용을 보고합니다.

예:

from typing import ClassVar


class Cat:
    color: ClassVar[str] = "white"
    weight: int

    def __init__(self, weight: int):
        self.weight = weight


Cat.color = "black"  # OK
my_cat = Cat(5)
my_cat.color = "gray"  # Error, setting class variable on instance

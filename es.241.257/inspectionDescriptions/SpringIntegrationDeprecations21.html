최신 버전의 Spring Integration XML 기반 애플리케이션 컨텍스트에서 지원이 중단된 XML 요소를 보고합니다.

자세한 내용은 마이그레이션 가이드를 참조하세요.

예:


<beans ... >
   <int:channel id="failChannel"
                dispatcher="failover"/> <!-- 이 속성은 지원 중단되었습니다. 디스패처의 하위 요소를 사용하세요 -->
   <int:poller default="true">
     <int:cron-trigger  expression="5"/>  <!-- 이 요소는 Spring Integration 2.1에서 제거되었습니다. 대신 'cron' 속성을 사용하세요 -->
     <int:interval-trigger interval="3"/> <!--  이 요소는 Spring Integration 2.1에서 제거되었습니다. 대신 interval-trigger 속성을 사용하세요 -->
  </int:poller>
  <ftp:inbound-channel-adapter session-factory="ftpSessionFactory"
                               cache-sessions="false" <!--  이 속성은 지원 중단되었습니다. 대신 CachingSessionFactory를 사용하세요 -->
                               channel="channel"/>
</beans>

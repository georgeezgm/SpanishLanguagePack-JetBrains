<bean> 속성의 구성 충돌을 보고합니다.

예:


  <beans>
    <bean class="java.lang.String" factory-bean="factoryBean"/>
    <!-- 'class' 또는 'factory-bean' 중 하나가 있어야 합니다 -->
    <bean factory-bean="factoryBean" id="missingFactoryMethod"/> <!--//factory-bean이 있으면, factory-method가 있어야 합니다 -->

    <!-- 디폴트 값 'false'가 있는 속성 abstract는 불필요합니다 -->/
    <bean abstract="false" id="superfluousDefaultAttributeAbstract"/>

    <!-- 디폴트 값 'singleton'이 있는 속성 scope는 불필요합니다 -->/
    <bean id="superfluousDefaultAttributeScope" scope="singleton"/>


    <bean class="java.lang.StringBuilder">
      <property name="length">
        <!-- 속성 ID, 이름, 범위는 불필요합니다 -->
        <bean id="beanId" name="beanName" scope="prototype"/>
      </property>
    </bean>
  </beans>

org.springframework.data.mongodb.repository.Query 어노테이션 대신 지정된 미해결 문서 필드를 보고합니다.

예:


@Document
class User {
  @Field
  String name;
}

interface UserRepository extends Repository<User, Long> {
  @Query("{ age: { $gt: ?0 } }") // 'age''가 미해결로 강조 표시됩니다
  List<User> findUsersOlderThan(Integer age);
}
